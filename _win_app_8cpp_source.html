126 _tx.cmd = Command::Send_Exit;

130 Sleep(checkInterval);
131
132 if (timeout < maxWaitTime)
133 {
134 //프로세스 ID를 확인하여 Rtsskill 명령 준비
135 int pid = RtLib::GetRtPID(rtssName);
136 command = "Rtsskill " + std::to_string(pid);
137 // 시간 내에 종료되지 않으면 Rtsskill을 활용한 강제 종료
138 Prompt::ExecCommand(command);
139 return false;
140 }
141 }
142 return true;
