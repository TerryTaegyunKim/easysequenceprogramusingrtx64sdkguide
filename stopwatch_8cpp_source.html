89 double Stopwatch::Update(int ch)
90 {
91 // 유효한 채널인지, 타이머가 동작 중인지 확인
92 if (ch < 0 || ch >= constants::MAX_TIMER_CHANNEL || !timeWatch[ch].isRunning)
93 return 0.0;
94
95 Timer& timer = timeWatch[ch];
96
97 // 타이머 종료 시간 설정 및 실패 시 0.0 반환
98#ifdef UNDER_RTSS
99 if (RtQueryPerformanceCounter(&timer.end) == 0) return 0.0;
100#else
101 if (QueryPerformanceCounter(&timer.end) == 0) return 0.0;
102#endif // UNDER_RTSS
103
104 // 경과 시간을 마이크로초 단위로 계산
105 double timeMicroseconds = static_cast<double>(timer.end.QuadPart - timer.start.QuadPart) * 1.0e6 / timer.freq.QuadPart;
106
107 // 타이머 통계 업데이트 (최소, 최대, 평균 시간 등)
108 if (timer.count == 0)
109 {
110 timer.min = timeMicroseconds;
111 timer.max = timeMicroseconds;
112 }
113 else
114 {
115 if (timeMicroseconds < timer.min) timer.min = timeMicroseconds;
116 if (timeMicroseconds > timer.max) timer.max = timeMicroseconds;
117 }
118
119 timer.total += timeMicroseconds;
120 timer.count++;
121 timer.cur = timeMicroseconds;
122 timer.avg = timer.total / timer.count;
123
124 // 다음 측정을 위해 타이머 시작 시간 리셋
125#ifdef UNDER_RTSS
126 if (RtQueryPerformanceCounter(&timer.start) == 0) return 0.0;
127#else
128 if (QueryPerformanceCounter(&timer.start) == 0) return 0.0;
129#endif // UNDER_RTSS
130
131 return timeMicroseconds;
132 }
