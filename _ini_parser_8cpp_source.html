168 bool IniParser::IsKeyMatching(const char* key, const char* refKey, int mode, char delimiter)
169 {
170 try
171 {
172 // key와 refKey가 nullptr이 아닌지 확인
173 if (key == nullptr || refKey == nullptr)
174 {
175 throw std::invalid_argument("Key or referenceKey is null.");
176 }
177
178 // 모드가 0인 경우, 전체 키를 직접 비교
179 if (mode == 0)
180 {
181 return strcmp(key, refKey) == 0;
182 }
183
184 // 모드가 1 또는 2인 경우
185 std::string keyStr(key);
186 std::string refKeyStr(refKey);
187
188 // 구분자를 기준으로 키와 기준 키를 분리
189 size_t keyPos = keyStr.find(delimiter);
190 size_t refKeyPos = refKeyStr.find(delimiter);
191
192 if (keyPos == std::string::npos || refKeyPos == std::string::npos)
193 {
194 throw std::invalid_argument("Delimiter not found in key or referenceKey.");
195 }
196
197 std::string keyBeforeDelimiter = keyStr.substr(0, keyPos);
198 std::string keyAfterDelimiter = keyStr.substr(keyPos + 1);
199 std::string refKeyBeforeDelimiter = refKeyStr.substr(0, refKeyPos);
200 std::string refKeyAfterDelimiter = refKeyStr.substr(refKeyPos + 1);
201
202 if (mode == 1)
203 {
204 // 모드 1: 구분자 앞부분 비교
205 return keyBeforeDelimiter == refKeyBeforeDelimiter;
206 }
207 else if (mode == 2)
208 {
209 // 모드 2: 구분자 뒷부분 비교
210 return keyAfterDelimiter == refKeyAfterDelimiter;
211 }
212 else
213 {
214 throw std::invalid_argument("Invalid mode specified.");
215 }
216 }
