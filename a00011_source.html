31 bool SharedMemory::AccessResource(std::function<void()> func)
32 {
33 bool attemptAccess = false;
34 constexpr DWORD milliseconds = 1000; // Timeout
35
36 // 세마포어를 기다립니다.
37 if (RtWaitForSingleObject(hSemaphore, milliseconds) == WAIT_OBJECT_0)
38 {
39
40 // 이벤트를 신호 상태로 설정합니다.
41 if (RtSetEvent(hEvent))
42 {
43 if (pSharedMemory)
44 {
45 func();
46 attemptAccess = true;
47 }
48 // 작업 완료 후 이벤트를 비신호 상태로 설정
49 RtResetEvent(hEvent);
50 }
51
52 // 세마포어 해제
53 RtReleaseSemaphore(hSemaphore, 1, NULL);
54 }
55 return attemptAccess;
56 }

58 bool SharedMemory::Open()
59 {
60 hSharedMemory = RtOpenSharedMemory(SHM_MAP_ALL_ACCESS, FALSE, constants::DEF_SHARED_MEMORY_AREA_NAME, reinterpret_cast<void**>(&pSharedMemory));
61 if (hSharedMemory == nullptr)
62 {
63 hSharedMemory = RtCreateSharedMemory(SHM_MAP_ALL_ACCESS, 0, sizeof(*pSharedMemory), constants::DEF_SHARED_MEMORY_AREA_NAME, reinterpret_cast<void**>(&pSharedMemory));
64 if (hSharedMemory == nullptr)
65 {
66 RtPrintf("DataManager Create LookUp Table SharedMemory Fail...\n");
67 return false;
68 }
69 // Initialize All Data Zero Set
70 memset(pSharedMemory, 0x00, sizeof(*pSharedMemory));
71 }
72
73 hSemaphore = RtOpenSemaphore(NULL, false, constants::DEF_SHARED_MEMORY_SEMAPHORE_NAME);
74 if (hSemaphore == nullptr)
75 {
76 hSemaphore = RtCreateSemaphore(NULL, 1, 1, constants::DEF_SHARED_MEMORY_SEMAPHORE_NAME);
77 if (hSemaphore == nullptr)
78 {
79 RtPrintf("CreatehSemaphore Fail...\n");
80 return false;
81 }
82 }
83
84 hEvent = RtOpenEvent(NULL, NULL, constants::DEF_SHARED_MEMORY_EVENT_NAME);
85 if (hEvent == nullptr)
86 {
87 hEvent = RtCreateEvent(NULL, TRUE, FALSE, constants::DEF_SHARED_MEMORY_EVENT_NAME);
88 if (hEvent == nullptr)
89 {
90 RtPrintf("RtCreateEvent Fail...\n");
91 return false;
92 }
93 }
94
95 return true;
96 }

98 bool SharedMemory::Close()
99 {
100 try
101 {
102
103 if (hSemaphore != nullptr && hSemaphore != INVALID_HANDLE_VALUE)
104 {
105 if (!RtCloseHandle(hSemaphore))
106 {
107 RtPrintf("Failed to close shared memory semaphore handle. Error: %d\n", GetLastError());
108 }
109 }
110
111 if (hEvent != nullptr && hEvent != INVALID_HANDLE_VALUE)
112 {
113 if (!RtCloseHandle(hEvent))
114 {
115 RtPrintf("Failed to close shared memory event handle. Error: %d\n", GetLastError());
116 }
117 }
118
119 if (hSharedMemory != nullptr && hSharedMemory != INVALID_HANDLE_VALUE)
120 {
121 if (!RtCloseHandle(hSharedMemory))
122 {
123 RtPrintf("Failed to close shared memory handle. Error: %d\n", GetLastError());
124 }
125 }
126
127 hSemaphore = nullptr;
128 hEvent = nullptr;
129 hSharedMemory = nullptr;
130 pSharedMemory = nullptr;
131
132 return true;
133 }
134 catch (const std::exception& ex)
135 {
136 RtPrintf("Exception occurred: %s\n", ex.what());
137 return false;
138 }

147 void SharedMemory::Send(SHMD tx)
148 {

151 if (platform == PlatformType::Windows)
