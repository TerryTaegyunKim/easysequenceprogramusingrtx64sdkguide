36

38 fSuccess = CreateProcessA(

40 (LPSTR)szCmdArg.c_str(), // command line
41 NULL, // process security attributes
42 NULL, // primary thread security attributes
43 TRUE, // TRUE=handles are inherited. Required.
44 CREATE_NEW_CONSOLE, // creation flags
45 NULL, // use parent's environment
46 NULL, // use parent's current directory
47 &si, // __in, STARTUPINFO pointer
48 &pi); // __out, receives PROCESS_INFORMATION

52 return szResult;
53 }
54
